\section{Orlicz 范数与次高斯}
\begin{definition}[Orlicz 范数]
随机变量 $Z$ 的 $\psi_2$ 范数定义为
\[
\|Z\|_{\psi_2} := \inf\{s>0:\ \E e^{Z^2/s^2}\le 2\}.
\]
\end{definition}

\begin{lemma}[基本性质（示例）]
若 $X$ 次高斯，则 $X^2-\E X^2$ 次指数。
\end{lemma}

\begin{KeyBox}
\textbf{要点}：\(\psi_2\)-有界 $\Rightarrow$ 指数尾界；平方中心化 $\Rightarrow$ \(\psi_1\)-有界。
\end{KeyBox}
